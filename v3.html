ESP32_DevKit.ino — enterDeepSleep()
// Clean radio teardown before deep sleep. From my ESP32 firmware.
void enterDeepSleep(uint32_t seconds) {
  saveState();                        // persist epoch + config to NVS
  if (g_nimbleInited) {
    NimBLEDevice::getAdvertising()->stop();
    NimBLEDevice::deinit(true);      // release the BLE controller
  }
  WiFi.mode(WIFI_OFF);
  esp_bt_controller_disable();
  delay(300);                       // let the 3.3 V rail recover
  esp_sleep_enable_timer_wakeup((uint64_t)seconds * 1000000ULL);
  esp_deep_sleep_start();             // ~64 mA -> ~6.9 mA (~89% less)
}
